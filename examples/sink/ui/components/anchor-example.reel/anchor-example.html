Anchor

The Anchor component wraps an <a> element and exposes its standard attributes as component properties.

Example - binding to Anchor properties

Link text: URL:

Motorola

  • HTML
    Link text: <input data-montage-id="linkText" autofocus type="text" name="" value="">
URL: <input data-montage-id="linkURL" type="text" name="" value="" placeholder="http://...">

<p><a href="http://www.motorola.com" data-montage-id="link1" target="_blank" icon="external">Motorola</a></p>
  • Serialization
    {
"link1" : {
   "prototype": "montage/ui/anchor.reel",
   "properties": {
       "element": {"#": "link1"}
   },
   "bindings": {
       "textContent": {"<<->": "@linkTextInput.value"},
       "href": {"<<->": "@urlTextInput.value"}
   }
},

"linkTextInput": {
   "prototype": "montage/ui/input-text.reel",
   "properties": {
       "element": {"#": "linkText"},
       "value": "Motorola"
   }
},

"urlTextInput": {
   "prototype": "montage/ui/input-text.reel",
   "properties": {
       "element": {"#": "linkURL"},
       "value": "http://www.motorola.com"
   }
}
}
  • CSS
    a[icon="external"] {
  padding-right: 20px;
  background: transparent url(../sink/images/external.gif) no-repeat center right;
  color: red;
}

Generating a list of links

This example generates uses the Anchor and Repetition components to generate a table of links.

Name Description
  • HTML
    <table data-montage-id="list1" class="bordered-table">
    <thead>
        <tr>
            <th>Name</th>
            <th>Description</th>
        </tr>
    </thead>
    <tbody data-montage-id="table-row">
        <tr>
            <td><a data-montage-id="link" target="_blank"></a></td>
            <td><span data-montage-id="desc"></span></td>
        </tr>
    </tbody>
</table>
  • Serialization
    {
    "data": {
        "prototype": "montage/ui/controller/array-controller",
        "properties": {
            "selectObjectsOnAddition": true,
            "automaticallyOrganizeObjects": true
        },
        "bindings": {
            "content": {"<-": "@owner.data"}
        }
    },

    "list1": {
        "prototype": "montage/ui/repetition.reel",
        "properties": {
            "element": {
                "#": "table-row"
            },
            "isSelectionEnabled": true,
            "contentController": {"@": "data"},
            "axis": "horizontal"
        }
    },

    "link": {
        "prototype": "montage/ui/anchor.reel",
        "properties": {
            "element": {"#": "link"}
        },
        "bindings": {
            "href": {"<-": "@list1.objectAtCurrentIteration.href"},
            "textContent": {"<-": "@list1.objectAtCurrentIteration.src"}
        }
    },

    "desc": {
        "prototype": "montage/ui/dynamic-text.reel",
        "properties": {
            "element": {"#": "desc"}
        },
        "bindings": {
            "value": {"<-": "@list1.objectAtCurrentIteration.description"}
        }
    },
