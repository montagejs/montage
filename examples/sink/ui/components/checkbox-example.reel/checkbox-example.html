Input Checkbox

The InputCheckbox component wraps an HTML checkbox input type with custom styles and functionality.

Examples

  • HTML
    <form class="form-stacked">
    <label class="checkbox inline"><input data-montage-id="checkbox1" type="checkbox" /> Milk</label>
    <label class="checkbox inline"><input data-montage-id="checkbox2" type="checkbox" /> Eggs</label>
</form>
  • Serialization

Binding the InputCheckbox to a Button's disabled state


  • HTML
    <form class="form-stacked">
    <label class="checkbox inline"><input type="checkbox" data-montage-id="agreeCheckbox"/> I agree to something</label><br/>
    <input type="submit" data-montage-id="continueButton" value="Continue" class="btn" disabled />
</form>
  • Serialization
    {
    "agreeCheckbox": {
        "prototype": "montage/ui/input-checkbox.reel",
        "properties": {
            "element": {"#": "agreeCheckbox"}
        }
    },
    "continueButton": {
        "prototype": "montage/ui/button.reel",
        "properties": {
            "element": {"#": "continueButton"}
        },
        "bindings": {
            "disabled": {"<-": "@agreeCheckbox.checked", "converter": {"@": "invertConverter"}}
        }
    },

    "invertConverter": {
        "prototype": "ui/components/converters.reel[InvertConverter]"
    }
}
  • JavaScript
    
var InvertConverter = exports.InvertConverter = Montage.create(Converter, {

   allowPartialConversion: {
       value: true
   },

   convert: {
       enumerable: false,
       value: function(v) {
           return !v;
       }
   },

   revert: {
       value: function(v) {
           return !v;
       }
   }
});
