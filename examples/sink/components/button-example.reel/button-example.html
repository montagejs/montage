Button

The Button component wraps a <button> or <input type="button"> element, and exposes the standard attributes of those elements as properties, which you can use with data binding, for example.

Examples

Button with an identifier used to link it to an handler method.

  • HTML
    <button class="text btn" data-montage-id="button1">Click Me</button>
  • Serialization
    {
    "button1": {
        "module": "montage/ui/button.reel",
        "name": "Button",
        "properties": {
            "element": {"#": "button1"},
            "identifier": "button1"
        },
        "listeners": [
            {
                "type": "action",
                "listener": {"@": "owner"}
            }
        ]
    },
    "logger": {
        "module": "logger.reel",
        "name": "Logger",
        "properties": {
            "element": {"#": "logger"}
        }
    },
    "owner": {
        "module": "components/button-example.reel",
        "name": "ButtonExample",
        "properties": {
            "element": {"#": "buttontest"},
            "logger": {"@": "logger"}
        }
    },

}
  • Javascript
    var Montage = require("montage/core/core").Montage,
    Component = require("montage/ui/component").Component;

exports.ButtonExample = Montage.create(Component, {

    handleButton1Action: {
        value: function() {
            this.log("Button - button1 - clicked");
        }
    },

    log: {
        value: function(msg) {
            this.logger.log(msg);
        }
    },

});

Submit Button without a specific handler. In this case, the handleAction() method defined in button-example.html will be called on when the "action" event is dispatched.

  • Serialization

        "name": "Button",
        "properties": {
            "element": {"#": "button2"}
        },
        "listeners": [
            {
                "type": "action",
                "listener": {"@": "owner"}
            }
        ]
    },
    "owner": {
        "module": "components/button-example.reel",
        "name": "ButtonExample",
        "properties": {
            "element": {"#": "buttontest"},
            "logger": {"@": "logger"}
        }
    }
}
  • Javascript
    var Montage = require("montage/core/core").Montage,
    Component = require("montage/ui/component").Component;

exports.ButtonExample = Montage.create(Component, {

    handleAction: {
        value: function() {
            this.log("Fallback action handler invoked as there is no specific handler for this button");
        }
    }

    log: {
        value: function(msg) {
            this.logger.log(msg);
        }
    },

});

Button groups

To create a button group, wrap the buttons in an parent element that applies appropriate CSS styles.

  • HTML
    <div class="btn-group">
    <button class="btn" data-montage-id="copy">Copy</button>
    <button class="btn" data-montage-id="cut">Cut</button>
    <button class="btn" data-montage-id="paste">Paste</button>
</div>

Adding an image icon

To add an image icon to a Button use CSS, either by setting a background image, or with the :before and :after pseudo elements with the content property:

  • HTML
    <button class="btn settings" data-montage-id="settings">Settings</button>
  • Serialization
    {
    "settingsButton": {
        "module": "montage/ui/button.reel",
        "name": "Button",
        "properties": {
            "element": {"#": "settings"},
            "identifier": "settings"
        },
        "listeners": [
            {
                "type": "action",
                "listener": {"@": "owner"}
            }
        ]
    },
    "owner": {
        "module": "components/button-example.reel",
        "name": "ButtonExample",
        "properties": {
            "element": {"#": "buttontest"},
            "logger": {"@": "logger"}
        }
    }
}
  • CSS
    .settings:before {
    content: url(images/settings.png);
    vertical-align: middle;
    display: inline-block;
    padding-right: 0.5em;
}

Handling mouse/touch events on a Button

When a event occurs on a component (or any element) which has registered an event listener for the event type, the Montage event manager infers what function handler on the listener object to call. This topic is covered in detail in Event Handling.

For example, consider the following serialized Button component. It declares an event listener for the button's "action" event, and also assigns the value "button1" to its "identifier" property. 

"button1": {
    "module": "montage/ui/button.reel",
    "name": "Button",
    "properties": {
        "element": {"#": "button1"},
        "identifier": "button1"
    },
    "listeners": [
        {
            "type": "action",
            "listener": {"@": "owner"}
        }
    ]
}

The specified listener object (owner) implements the following function, which is automatically invoked when the Button dispatches the "action event."

handleButton1Action: {
    value: function() {
        this.log("Button - button1 - clicked");
    }
}

For more information, see Event Handling.
