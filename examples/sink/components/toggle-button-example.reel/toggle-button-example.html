Toggle Button

The ToggleButton component exposes can switch between pressed and unpressed state.

Demo

  • HTML

    This example's markup defines three <button> elements.

    <div class="btn-group font-styles">
    <button class="btn font-styles" id="bold">B</button>
    <button class="btn font-styles" id="italic" >I</button>
    <button class="btn font-styles" id="underline" >U</button>
</div>
<div><input type="text" id="display" value="This is some example text"></div>
  • Serialization

    The serialization associates each of the three <button> elements with the ToggleButton 

    {
   "bold": {
        "prototype": "montage/ui/toggle-button.reel",
        "properties": {
            "element": {"#": "bold"},
            "pressedClass": "active"
        }
    },
    "italic": {
        "prototype": "montage/ui/toggle-button.reel",
        "properties": {
            "element": {"#": "italic"},
            "pressedClass": "active"
        }
    },
    "underline": {
        "prototype": "montage/ui/toggle-button.reel",
        "properties": {
            "element": {"#": "underline"},
            "pressedClass": "active"
        }
    },
    "display": {
        "prototype": "components/style-element.reel",
        "properties": {
            "element": {"#": "display"}
        },
        "bindings": {
            "bold": {"<-": "@bold.pressed"},
            "italic": {"<-": "@italic.pressed"},
            "underline": {"<-": "@underline.pressed"}
        }
    }
}
  • Javascript: StyleElement
    var Montage = require("montage/core/core").Montage,
    Component = require("montage/ui/component").Component;

exports.StyleElement = Montage.create(Component, {
    hasTemplate: {value: false},

    bold: {
        get: function() {
            return this.element.style.fontWeight === "bold";
        },
        set: function(val) {
            this.element.style.fontWeight = (val) ? "bold" : "normal";
        }
    },
    underline: {
        get: function() {
            return this.element.style.textDecoration === "underline";
        },
        set: function(val) {
            this.element.style.textDecoration = (val) ? "underline" : "none";
        }
    },
    italic: {
        get: function() {
            return this.element.style.fontStyle === "italic";
        },
        set: function(val) {
            this.element.style.fontStyle = (val) ? "italic" : "normal";
        }
    }
});

ToggleSwitch component

The ToggleSwitch component provides a styled toggle control common to touch interfaces. This component operates on a standard <div> element, rather than a native HTML control.

  • HTML
    <div id="toggle"></div>
  • Serialization
    {
    "toggle": {
        "prototype": "montage/ui/toggle-switch.reel",
        "properties": {
            "element": {"#": "toggle"}
        }
    }
}
