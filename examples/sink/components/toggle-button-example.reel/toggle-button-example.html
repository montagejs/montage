Toggle Button

The ToggleButton component exposes can switch between pressed and unpressed state.

Demo

  • HTML

    This example's markup defines three <button> elements.

    <div class="montage-buttonGroup font-styles">
    <button class="font-styles" data-montage-id="bold">B</button>
    <button class="font-styles" data-montage-id="italic" >I</button>
    <button class="font-styles" data-montage-id="underline" >U</button>
</div>
<input type="text" data-montage-id="display" value="This is some example text">
  • Serialization

    The serialization associates each of the three <button> elements with the ToggleButton

  • JavaScript: StyleElement
    var Montage = require("montage/core/core").Montage,
    Component = require("montage/ui/component").Component;

exports.StyleElement = Montage.create(Component, {
    hasTemplate: {value: false},

    _bold: {
        value: false
    },
    bold: {
        get: function() {
            return this._bold;
        },
        set: function(val) {
            this._bold = !!val;
            this.needsDraw = true;
        }
    },
    _underline: {
        value: false
    },
    underline: {
        get: function() {
            return this._underline;
        },
        set: function(val) {
            this._underline = !!val;
            this.needsDraw = true;
        }
    },
    _italic: {
        value: false
    },
    italic: {
        get: function() {
            return this._italic;
        },
        set: function(val) {
            this._italic = !!val;
            this.needsDraw = true;
        }
    },

    draw: {
        value: function() {
            this.element.style.fontWeight = (this._bold) ? "bold" : "normal";
            this.element.style.textDecoration = (this._underline) ? "underline" : "none";
            this.element.style.fontStyle = (this._italic) ? "italic" : "normal";
        }
    }
});

ToggleSwitch component

The ToggleSwitch component provides a styled toggle control common to touch interfaces. This component operates on a standard <div> element, rather than a native HTML control.

  • HTML
    <div data-montage-id="toggle"></div>
  • Serialization
    {
    "toggle": {
        "prototype": "montage/ui/toggle-switch.reel",
        "properties": {
            "element": {"#": "toggle"}
        }
    }
}
