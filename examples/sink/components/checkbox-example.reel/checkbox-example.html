Checkbox

The Checkbox component wraps an HTML checkbox input type with custom styles and functionality.

Examples

  • html 
    <form class="form-stacked">
    <label class="checkbox inline"><input id="checkbox1" type="checkbox" /> Milk </label>
    <label class="checkbox inline"><input id="checkbox2" type="checkbox" /> Eggs</label>
</form>
  • serialization 
    {
    "checkbox1": {
        "module": "montage/ui/checkbox.reel",
        "name": "Checkbox",
        "properties": {
            "element": {"#": "checkbox1"}

            {
                "type": "action",

Binding the Checkbox to a Button's disabled state


  • html 
    <form class="form-stacked">
    <label class="checkbox inline"><input type="checkbox" id="agree-checkbox"/> I agree to something</label><br/>
    <input type="submit" id="continue-button" value="Continue" class="btn" disabled />
</form>
  • serialization 
    {
    "agree-checkbox": {
        "module": "montage/ui/checkbox.reel",
        "name": "Checkbox",
        "properties": {
            "element": {"#": "agree-checkbox"}
        }
    },
    "continue-button": {
        "module": "montage/ui/button.reel",
        "name": "Button",
        "properties": {
            "element": {"#": "continue-button"}
        },
        "bindings": {
            "disabled": {
                "boundObject": {"@": "agree-checkbox"},
                "boundObjectPropertyPath": "checked",
                "converter": {"@": "invert-converter"},
                "oneway": true
            }
        }
    },

    "invert-converter": {
        "module": "components/converters.reel",
        "name": "InvertConverter"
    }
}
  • javascript 
    var InvertConverter = exports.InvertConverter = Montage.create(Converter, {

    allowPartialConversion: {
        value: true
    },

    convert: {
        enumerable: false,
        value: function(v) {
            return !v;
        }
    },

    revert: {
        value: function(v) {
            return !v;
        }
    }
});
