Checkbox

The Checkbox component wraps an HTML checkbox input type with custom styles and functionality.

Serialization
"checkbox1": {
    "module": "montage/ui/bluemoon/checkbox.reel",
    "name": "Checkbox",
    "properties": {
        "element": {"#": "checkbox1"}
    }
}
HTML Markup
<label> <input id="checkbox1" type="checkbox" /> Milk</label>

Binding


Serialization
"agree-checkbox": {
    "module": "montage/ui/checkbox.reel",
    "name": "Checkbox",
    "properties": {
        "element": {"#": "agree-checkbox"}
    }
},
"continue-button": {
    "module": "montage/ui/button.reel",
    "name": "Button",
    "properties": {
        "element": {"#": "continue-button"}
    },
    "bindings": {
        "disabled": {
            "boundObject": {"@": "agree-checkbox"},
            "boundObjectPropertyPath": "checked",
            "converter": {"@": "invert-converter"},
            "oneway": true
        }
    }
},

"invert-converter": {
    "module": "components/converters.reel",
    "name": "InvertConverter"
}
HTML Markup
<label><input type="checkbox" id="agree-checkbox"/> I agree to something</label>

<input type="submit" id="continue-button" value="Continue" class="btn" disabled />
