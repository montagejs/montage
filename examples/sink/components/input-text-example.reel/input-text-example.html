Textfield

The Textfield component wraps an HTML <input> text field. The example uses data binding to echo the text you enter in a <p> element, which is used to represent a DynamicText component.

HTML
<input type="text" id="text1" />
<p id="echo1"></p>
Serialization
"text1": {
    "module": "montage/ui/textfield.reel",
    "name": "Textfield",
    "properties": {
        "element": {"#": "text1"},
        "value": "Enter some text..."
    }
},

"echo1": {
    "module": "montage/ui/dynamic-text.reel",
    "name": "DynamicText",
    "properties": {
        "element": {"#": "echo1"}
    },
    "bindings": {
        "value": {
            "boundObject": {"@": "text1"},
            "boundObjectPropertyPath": "value"
        }
    }
}

TextArea

The TextArea component wraps an HTML <textarea> element.

HTML
<textarea id="textarea1"></textarea>
<p id="echo2"></p>
Serialization
"textarea1": {
    "module": "montage/ui/TextArea.reel",
    "name": "TextArea",
    "properties": {
        "element": {"#": "textarea1"}
        "value": "Enter some text..."
    }
},
"echo2": {
    "module": "montage/ui/dynamic-text.reel",
    "name": "DynamicText",
    "properties": {
        "element": {"#": "echo2"}
    },
    "bindings": {
        "value": {
            "boundObject": {"@": "textarea1"},
            "boundObjectPropertyPath": "value"
        }
    }
}
