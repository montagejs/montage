Scroller

The Scroller component provides vertical and horizontal scrolling of fixed height elements. It supports momentum and bounce behaviors that you can control or disable.

    {

    "owner": {
        "prototype": "components/scroller-example.reel",
        "properties": {
            "element": {"#": "scrollview-example"}
        }
    },

    "scroller": {
        "prototype": "montage/ui/scroller.reel",
        "properties": {
            "element": {"#": "scroller"}
        },
        "bindings": {
            "hasMomentum": {
                "boundObject": {"@": "toggleMomentum"},
                "boundObjectPropertyPath": "checked"
            },
            "hasBouncing": {
                "boundObject": {"@": "toggleBouncing"},
                "boundObjectPropertyPath": "checked"
            }
        }
    },

    "toggleMomentum": {
        "prototype": "montage/ui/checkbox.reel",
        "properties": {
            "element": {"#": "toggleMomentum"},
            "checked": true
        }
    },
    "toggleBouncing": {
        "prototype": "montage/ui/checkbox.reel",
        "properties": {
            "element": {"#": "toggleBouncing"},
            "checked": true
        }
    }
}
