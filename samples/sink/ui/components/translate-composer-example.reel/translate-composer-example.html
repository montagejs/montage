TranslateComposer

The TranslateComposer listens for touch/mouse events on an element or component and then calculates properties based on the mouse movement, such as momentum. You can then bind these calculated properties to another element or component, such as a slider control, for example.The following example uses the TranslateComposer to drag an element around the screen.

  • HTML
    <div class="wrap">
    <div class="mover" data-montage-id="example">Drag me</div>
</div>

<div class="controls">
    Axis:
    <select data-montage-id="axis" required>
        <option value="both" selected="true">Both</option>
        <option value="horizontal" >Horizontal</option>
        <option value="vertical">Vertical</option>
    </select>
    <div class="momentum">
        <label for="cb1"><input type="checkbox" data-montage-id="checkboxMomentum" data-montage-id="cb1" id="cb1" /> Momentum </label>
    </div>
    <code>translateX: <span data-montage-id="x_value"></span></code>
    <code>translateY: <span data-montage-id="y_value"></span></code>
</div>
  • Serialization
    {
    "translate_composer": {
        "prototype": "montage/ui/composer/translate-composer",
        "properties": {
            "component": {"@": "example"},
            "maxTranslateX": 350,
            "maxTranslateY": 150,
            "pointerSpeedMultiplier": -1
        },
        "bindings": {
            "axis": {"<-": "@axisSelect.value"},
            "hasMomentum": {"<<->": "@checkboxMomentum.checked"}
        }
    },

    "example": {
        "prototype": "montage/ui/dynamic-text.reel",
        "properties": {
            "element": {"#": "example"},
            "value": "Drag me"
        }
    },

    "owner": {
        "prototype": "ui/components/translate-composer-example.reel",
        "properties": {
            "element": {"#": "example-div"},
            "translate_composer": {"@": "translate_composer"},
            "example": {"@": "example"}
        },
        "bindings": {
            "x": {"<-": "@translate_composer.translateX"},
            "y": {"<-": "@translate_composer.translateY"}
        }
    },

    "axisSelect": {
        "prototype": "montage/ui/select.reel",
        "properties": {
            "element": {"#": "axis"}
        }
    },

    "checkboxMomentum": {
        "prototype": "montage/ui/input-checkbox.reel",
        "properties": {
            "element": {"#": "checkboxMomentum"},
            "checked": true
        }
    },

    "xvalue": {
        "prototype": "montage/ui/dynamic-text.reel",
        "properties": {
            "element": {"#": "x_value"}
        },
        "bindings": {
            "value": {"<<->": "@translate_composer.translateX"}
        }
    },

    "yvalue": {
        "prototype": "montage/ui/dynamic-text.reel",
        "properties": {
            "element": {"#": "y_value"}
        },
        "bindings": {
            "value": {"<<->": "@translate_composer.translateY"}
        }
    },
    "translateAccordion1": {
        "prototype": "ui/accordion.reel",
        "properties": {
            "element": {"#": "translateAccordion1"}
        }
    }
}
